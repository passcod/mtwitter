Parser.prototype.checkHeartbeat = function(){
  var self = this;
  var currentTime = new Date().getTime();

  if (self.lastTime !== null && self.lastTime < (currentTime - 30000)) {
    self.emit('missedHeartbeat');
  }

  setTimeout(function() {
    self.checkHeartbeat();
  }, 30000);
};
